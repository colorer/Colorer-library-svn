  4: global proc int getDataWord( int $fid)
  5: {
  6: global    int    $readFromOd;
  7: int    $value;
  8: 
  9:     if (!$readFromOd) {
 10:         fgetword( $fid);
 11:         $readFromOd = 1;
 12:     }
 13:     $value = fgetword( $fid);
 14:     $readFromOd = ($readFromOd==8 ? 0 : $readFromOd+1);
 15:     return( $value);
 16: }

 33: int    $dGroupSize = $dSlotSize * $dataSlots;
 34: int    $readOffset = $readDataGroup * $dGroupSize + $readSlot * $dSlotSize;
 35:

 43: float    $pLY = (($gridResY * $pSpacing) - $pSpacing) / 2;
 44: 
 45: // Entsprechendes Partikel Objekt herstellen
 46: 
 47: string $pObjectName[] = `particle -grs $pSpacing -ll (-$pLX) 0 (-$pLY) -ur $pLX 0 $pLY`;
 48: addAttr -ln "rgbPP" -dt vectorArray $pObjectName[1];
 49: addAttr -ci true -ln "rgbVPP0" -sn "rgbVPP0" -dt "doubleArray" $pObjectName[1];
 50:

 52: 
 53: $readFromOd = 0;
 54: int $fid = popen( $pipeCmd, "r");
 55: int $dSlotWords = $gridResX * $gridResY;
 56: //int $dSlotWords = 1000;
 57: float    $value[];
 58: float    $minVal = 1000;
 59: float    $maxVal = -1000;
 60: float    $q;
 61: string $xxx;

 63:

 66: $maxVal = max( $maxVal, $value[$i]);
 67: $minVal = min( $minVal, $value[$i]);
 68: }
 69: pclose( $fid);
 70: print("Ende Lesen ... " + `system( "date")` + "\n");
 71:

 73: print( "Max Temp: " + $maxVal + "\n");
 74: $v = ($maxVal-$minVal);
