0: function phone(action)
  1: %PHONE  Signal processing and the touch-tone phone.
  2: 
  3: %   Author(s): Ned Gulley, 6-21-93
  4: %   Copyright 1988-2000 The MathWorks, Inc.
  5: %   $Revision$  $Date$
  6: 
  7: persistent Fs tones t;
  8: if isempty(tones),
  9:     % Generate DTMF tones:
 10:     Fs  = 8000;
 11:     t   = (0:799)/Fs;    % 800 samples at Fs
 12:     pit = 2*pi*t;
 13:     
 14:     fc = [697 770 852 941];
 15:     fr = [1209 1336 1477]; 
 16:     f  = [];
 17:     for c=1:4,
 18:         for r=1:3,
 19:             f = [ f [fc(c);fr(r)] ];
 20:         end
 21:     end
 22:     
 23:     for i=1:size(f,2),
 24:         tones(:,i) = sum(sin(f(:,i)*pit))' / 4;
 25:     end
 26: end
 27: 
 28: if nargin<1,
 29:     action='initialize';
 30: end
 31: 
 32: if strcmp(action,'initialize'),
 33:     figNumber=figure( ...
 34:         'Name','Phone Pad', ...
 35:         'NumberTitle','off', ...
 36:         'Backingstore','off', ...
 37:         'Visible','off');
 38:     axHndl1=axes( ...
 39:         'Units','normalized', ...
 40:         'Position',[0.10 0.60 0.60 0.32], ...
 41:         'Drawmode','fast', ...
 42:         'Visible','on');
 43:     axHndl2=axes( ...
 44:         'Units','normalized', ...
 45:         'Position',[0.10 0.10 0.60 0.32], ...
 46:         'Drawmode','fast', ...
 47:         'Visible','on');
 48: 
 49:     %===================================
 50:     % Information for all buttons
 51:     labelColor=[0.8 0.8 0.8];
 52:     top=0.85;
 53:     left=0.75;
 54:     bottom=0.05;
 55:     xLabelPos=0.75;
 56:     labelWid=0.20;
 57:     labelHt=0.05;
 58:     btnWid=0.20;
 59:     btnHt=0.08;
 60:     % Spacing between the button and the next command's label
 61:     spacing=0.05;
 62:     btnOffset=0;
 63:     
 64:     %====================================
 65:     % The CONSOLE frame
 66:     % Use the frame's userdata to hold the tones matrix
 67:     frmBorder=0.02;
 68:     yPos=bottom-frmBorder;
 69:     frmPos=[left-frmBorder yPos btnWid+2*frmBorder 0.9+2*frmBorder];
 70:     frmHndl=uicontrol( ...
 71:         'Style','frame', ...
 72:         'Units','normalized', ...
 73:         'Position',frmPos, ...
 74:         'BackgroundColor',[0.50 0.50 0.50]);
 75: 
 76:     %====================================
 77:     % The TONE buttons
 78:     %btnNumber=1;
 79:     %yLabelPos=0.90-(btnNumber-1)*(btnHt+labelHt+spacing);
 80:     labelStr=' Button No.';
 81:     
 82:     for count=1:12,
 83:         btnPos=[xLabelPos+rem(count-1,3)*(btnWid/3) ...
 84:                 top-floor((count-1)/3)*(btnWid/3) btnWid/3 btnWid/3];
 85:         if count<10,
 86:             btnStr=num2str(count);
 87:         else
 88:             s = '*0#';
 89:             btnStr = s(count-9);
 90:         end
 91:         uicontrol( ...
 92:             'Style','pushbutton', ...
 93:             'Units','normalized', ...
 94:             'Position',btnPos, ...
 95:             'String',btnStr, ...
 96:             'UserData',count, ...
 97:             'Callback','phone(''tone'')');
 98:     end
 99: 
100:     %====================================
101:     % The SOUND button
102:     btnNumber=2;
103:     yPos=top-3*btnHt;
104:     labelStr='Sound';
105:     % Setting this checkbox will allow the sound to work
106:     
107:     % Generic button information
108:     btnPos=[left yPos-btnHt btnWid btnHt];
109:     sndHndl=uicontrol( ...
110:         'Style','checkbox', ...
111:         'Units','normalized', ...
112:         'Position',btnPos, ...
113:         'Enable','on', ...
114:         'String',labelStr);
115: 
116:     %====================================
117:     % The INFO button
118:     labelStr='Info';
119:     callbackStr='phone(''info'')';
120:     infoHndl=uicontrol( ...
121:         'Style','push', ...
122:         'Units','normalized', ...
123:         'Position',[left bottom+btnHt+spacing btnWid btnHt], ...
124:         'String',labelStr, ...
125:         'Callback',callbackStr);
126: 
127:     %====================================
128:     % The CLOSE button
129:     labelStr='Close';
130:     callbackStr='close(gcf)';
131:     closeHndl=uicontrol( ...
132:         'Style','push', ...
133:         'Units','normalized', ...
134:         'Position',[left bottom btnWid btnHt], ...
135:         'String',labelStr, ...
136:         'Callback',callbackStr);
137: 
138:     % Initialize the plot with tone number 1
139:     axes(axHndl1);
140:     tone=tones(:,1);
141:     plot1Hndl=plot(t,tone,'Erasemode','background');
142:     xlabel('Time (sec)');
143:     ylabel('Signal');
144:     title('Time Response');
145:     set(axHndl1,'XLim',[0 0.05],'YLim',[-1 1]);
146:     axes(axHndl2);
147:     p = psd(tones(:,1),256,[],[],128);
148:     f = (0:127)/128 * (Fs/2);
149:     f=f(1:63);
150:     p=p(1:63,1);
151:     plot2Hndl=semilogy(f,p,'Erasemode','background');
152:     xlabel('Frequency (Hz)');
153:     ylabel('Signal Power');
154:     title('Spectrum');
155:     set(axHndl2,'XLim',[0 2000]);
156: 
157:     % Uncover the figure
158:     set(figNumber,'Visible','on', ...
159:         'UserData',[frmHndl infoHndl closeHndl plot1Hndl plot2Hndl sndHndl]);
160: 
161: elseif strcmp(action,'tone'),
162:     figNumber=watchon;
163:     hndlList=get(figNumber,'Userdata');
164:     frmHndl=hndlList(1);
165:     infoHndl=hndlList(2);
166:     closeHndl=hndlList(3);
167:     plot1Hndl=hndlList(4);
168:     plot2Hndl=hndlList(5);
169:     sndHndl=hndlList(6);
170:     %set([closeHndl infoHndl],'Enable','off');
171: 
172:     % ====== Playback the tone here
173:     % Sampling rate is Fs Hz
174:     toneChoice=get(gco,'UserData');
175:     tone=tones(:,toneChoice);
176:     if get(sndHndl,'Value'),
177:         sound(tone,Fs);
178:     end
179:     p = psd(tone,256,[],[],128);
180:     f = (0:127)/128 * (Fs/2);
181:     p=p(1:63,1);
182:     set(plot1Hndl,'YData',tone);
183:     set(plot2Hndl,'Ydata',p);
184:     % ====== End of playing tone
185: 
186:     %set([closeHndl infoHndl],'Enable','on');
187:     watchoff(figNumber);
188: 
189: 
190: elseif strcmp(action,'info'),
191:     ttlStr=get(gcf,'Name');
192:     hlpStr= ...                                          
193:         [' This window demonstrates the speed and     '  
194:          ' utility of the "psd" command in the        '  
195:          ' Signal Processing Toolbox.                 '  
196:          '                                            '  
197:          ' The touch tone phone pad in the upper right'  
198:          ' plays the actual tones used by a normal    '  
199:          ' phone. The upper plot shows a sample of the'  
200:          ' time response of the sound. The lower plot '  
201:          ' shows the spectrum associated with the     '  
202:          ' waveform.                                  '  
203:          '                                            '  
204:          ' Notice that two tones are used to code each'  
205:          ' number on the dial.                        '  
206:          '                                            '  
207:          ' File name: phone.m                         '];
208:     helpfun(ttlStr,hlpStr);                              
209: 
210: end    % if strcmp(action, ...
211: 
212: % [EOF] phone.m
