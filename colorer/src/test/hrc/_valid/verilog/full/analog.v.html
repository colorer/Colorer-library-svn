0: //title netlist for analog to digital converter
  1: // file adc_gate.v
  2: /*      This file demonstrates the following capabilities for SILOS III for
  3:     mixed analog and digital simulation:
  4:     -  passing analog values in module ports
  5:     -  modeling a simple analog comparator
  6:     -  modeling a simple digital to analog converter
  7:     -  nested `define macros (OVI 2.0 feature)
  8:     -  UDP primitives for D flip-flop and JK flip-flop
  9: */
 10: 
 11: !con .ext=all        // This is a switch to allow the SILOS III analog extensions
 12: 
 13: `define abs1(a) (((a)<0)?-(a):(a))
 14: `define close(a,b) (`abs1(a-b) < 0.01) ? 1:0
 15: `define high  'b1                       
 16: `define low   'b0
 17: 
 18: // this module defines the stimulus and instantiates the circuit
 19: 
 20: `timescale 1ns / 1ns
 21: 
 22: module top;
 23:     wire     up, down;       
 24:     real  analog_in;
 25:     wire real  feedback;
 26:     wire [7:0] count;
 27:     reg [7:0] in;
 28:     reg     clock;
 29:     wire    down_up;
 30:     wire    enable;
 31:     reg     load_inputs;
 32:     reg     clear;
 33:     integer counter_value;
 34: 
 35:     comparator comp(analog_in,feedback,up,down);
 36:     dac dig2an(feedback, count);
 37:     sr_flop_gate sr_flop(up, down, clock, down_up, clear, enable);
 38:     counter_gate counter (in, clock, down_up, enable, load_inputs, count);
 39: 
 40:     initial
 41:         begin
 42:         clock = 'b0;
 43:         forever
 44:             begin
 45:             #2000 clock = 'b1;
 46:             #2000 clock = 'b0;
 47:             #500 clock = 'b1;
 48:             end
 49:         end
 50:     initial
 51:         fork
 52:                  in = 'b0;
 53: 
 54:                  clear = 'b0;
 55:             #100 clear = 'b1;
 56: 
 57:                  load_inputs = 'b1;
 58:             #1   load_inputs = 'b0;
 59:             #600 load_inputs = 'b1;
 60: 
 61:                  analog_in = 0.0;
 62:             #1k  analog_in = 0.5;
 63:             #30k analog_in = 1.0;
 64:             #50k analog_in = 1.8;
 65:             #90k analog_in = 1.0;
 66:             #130k analog_in = 0.5;
 67:             #150k analog_in = 0.0;
 68:             #180k $finish;
 69:         join
 70:     always @count counter_value = integer(feedback*10);
 71: endmodule
 72: 
 73: // this module models an analog comparator
 74: module comparator(analog_in,feedback,up,down);  // analog signals are passed in
 75:     input analog_in,feedback;
 76:     output up, down;
 77:     real analog_in,feedback;                // ports (analog_in, feedback)
 78:     reg up, down;
 79:     always
 80:         begin
 81:             @ (analog_in or feedback);
 82:             up = `low;
 83:             down = `low;
 84:             if (!`close(analog_in,feedback))
 85:                 begin
 86:                 if ( analog_in > feedback )
 87:                     up = `high;
 88:                 else if ( analog_in < feedback )
 89:                     down = `high;
 90:                 end
 91:         end
 92: endmodule
 93: 
 94: //this module models a digital to analog converter
 95: module dac(feedback, count);
 96:     input[7:0] count;
 97:     output feedback;
 98:     real feedback;
 99:     initial
100:         feedback = 0.0;
101:     always
102:         @ (count) #121 feedback = real(integer(count)) / 10.0;
103: endmodule
104: 
105: //  Gate level latching logic
106: module sr_flop_gate (up, down, clock, down_up, clear, enable);
107:     // this module models latching circuitry at the gate level:
108:     supply1 vdd;
109:     input up, down, clock, clear;
110:     output down_up, enable;
111:     wire up, down, clock, clear;
112:     wire down_up, enable;
113:     and     #(320, 280) 
114:         (and1, clock, up),
115:         (and2, clock, down),
116:         (and3, ff1_q, ff2_qb),
117:         (down_up, ff2_q, ff1_qb);
118:     and     #330
119:         (clock_d, clock);
120:     nor     #150 nor1 (enable, down_up, and3);
121:     
122:     d_flip_flop     #(220, 210)
123:             (ff1_q, clock_d, and1, vdd, clear),
124:             (ff2_q, clock_d, and2, vdd, clear);
125:     not     (ff1_qb,ff1_q),
126:         (ff2_qb,ff2_q);
127: endmodule
128: 
129: // ls191 up-down counter
130: module ls191 (d,b,c,a, clk, du, ct, ld, od, oc, ob, oa, rco, mo);
131:     input a,b,c,d,clk,du,ct,ld;
132:     output oa,ob,oc,od,rco,mo;
133:     wire clk;
134:     wire a,b,c,d,du,ct,ld;
135:     wire s0, s1, s2, s3;
136:     wire c0, c1, c2, c3;
137:     
138:     jk_flip_flop
139:         (q0, ck, ~ct, ~ct, s0, c0),
140:         (q1, ck, ~d1, ~d1, s1, c1),
141:         (q2, ck, ~d2, ~d2, s2, c2),
142:         (q3, ck, ~d3, ~d3, s3, c3);
143:     not     (q0n,q0),
144:         (q1n,q1),
145:         (q2n,q2),
146:         (q3n,q3);
147:     and     #(240, 360)
148:         (oa, q0),
149:         (ob, q1),
150:         (oc, q2),
151:         (od, q3);
152:     // rc output
153:     not     #(210, 250) (tcd, tc);
154:     nand    #(200, 240) (rco, clk, ctnd, tcd);
155:     // tc output
156:     and     #(90, 190)
157:         (n53, q0, q1, q2, q3),
158:         (n54,  q0n, q1n, q2n, q3n);
159:     //aoi   
160:     assign  tc = (~(((~du) & n53) | (du & n54)));
161:     and     #330 (mo, ~tc);
162:     // other logic
163:     nor
164:         (cue, du, ct),
165:         (ce, ~du, ct);
166:     nand    
167:         (s0, ad, ldnd),
168:         (c0, s0, ldne),
169:         (s1, bd, ldnd),
170:         (c1, s1, ldne),
171:         (s2, cd, ldnd),
172:         (c2, s2, ldne),
173:         (s3, dd, ldnd),
174:         (c3, s3, ldne);
175:     
176:     // aoi logic
177:     assign  d1 = (~((ce & q0n) | ( q0 & cue)));
178:     assign  d2 = (~((ce & q0n & q1n) | (q0 & q1 & cue)));
179:     assign  d3 = (~((ce & q0n & q1n & q2n) | (q0 & q1 & q2 & cue)));
180:     not
181:          (ck, clk);
182:     and     #(80,40)
183:          (ad, a),
184:         (bd, b),
185:         (cd, c),
186:         (dd, d);
187:     not     #(90,130)
188:         (ctnd, ct);
189:     not     #(90,1)
190:         (ldnd, ld);
191:     not     #(40,1)
192:         (ldne, ld);
193: endmodule

206:         ls191_2 (in[7],in[6],in[5],in[4], clock, down_up, ripple_clk, 
207:         load_inputs,  count[7],count[6],count[5],count[4], foo,foo);

210: // d flip-flop udp primitive
211: primitive d_flip_flop (q, clock, data, set, clear);
212:     input clock, data, set, clear;
213:     output q;
214:     reg q;
215: 
216:     table
217:     //      clock   data    set     clear   q       q+
218:         ?       ?       0       ?       :?:     1;
219:         ?       ?       x       0       :?:     0;
220:         ?       ?       1       0       :?:     0;
221:         (01)    0       1       1       :?:     0;
222:         (01)    1       1       1       :?:     1;
223:         (01)    x       1       1       :?:     x;
224:         f       ?       1       1       :?:     -;
225:         ?       ?       1       1       :?:     -;
226:     endtable
227: endprimitive

229: // jk flip-flop udp primitive
230: primitive jk_flip_flop (q, clock, j,k, set, clear);
231:     input clock, j,k, set, clear;
232:     output q;
233:     reg q;
234:     initial  q = 1'b0;
235:     table
236:     //      clock   jk      set     clear   q       q+
237:         ?       ??      0       ?       :?:     1;
238:         ?       ??      x       0       :?:     0;
239:         ?       ??      1       0       :?:     0;
240:         (01)    00      1       1       :0:     0;
241:         (01)    00      1       1       :x:     x;
242:         (01)    00      1       1       :1:     1;
243:         (01)    10      1       1       :?:     1;
244:         (01)    01      1       1       :?:     0;
245:         (01)    11      1       1       :1:     0;
246:         (01)    11      1       1       :x:     x;
247:         (01)    11      1       1       :0:     1;
248:         (01)    x?      1       1       :?:     x;
249:         (01)    ?x      1       1       :?:     x;
250:         f       ??      1       1       :?:     -;
251:         ?       ??      1       1       :?:     -;
252:     endtable
253: endprimitive
